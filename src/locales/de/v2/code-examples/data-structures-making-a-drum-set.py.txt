# Erstellen eines Drum Sets: Benutze Listen mit makeBeat()

# Setup
from earsketch import *
setTempo(100)

# Vorher hatten wir eine Spur für jeden Sound (measure 1):
kick = OS_KICK05
snare = OS_SNARE01
kickBeat = "0+++----0+++----"
snareBeat = "----0+++----0+++"
makeBeat(kick, 1, 1, kickBeat)
makeBeat(snare, 2, 1, snareBeat)

# Jetzt können wir sie kombinieren (measure 3):
drums = [OS_KICK05, OS_SNARE01]
beat = "0+++1+++0+++1+++"
makeBeat(drums, 1, 3, beat)
