# Benutzereingabe 1: Den Benutzer nach dem Tempo fragen

# Setup
from earsketch import *

# nach Tempo fragen
question = "Welches Tempo möchtest du für deine Musik? Wähle eine Zahl zwischen 45 und 220"
answer = readInput(question)

# in eine Ganzzahl konvertieren
tempo = int(answer)

# Tempo festlegen
setTempo(tempo)

# Musik
fitMedia(COMMON_LOVE_THEME_STRINGS_1, 1, 1, 5)
