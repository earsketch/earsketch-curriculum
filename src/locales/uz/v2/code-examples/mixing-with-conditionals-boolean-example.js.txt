// Mantiqiy misol: Treklarimizning ovozini balandligin tahlil qilamiz
// Sozlash
setTempo(120);

// 2ta trekni yaratish
var melody1 = RD_CINEMATIC_SCORE_STRINGS_14;
var melody2 = RD_UK_HOUSE__5THCHORD_1;
fitMedia(melody1, 1, 1, 9);
fitMedia(melody2, 2, 1, 9);

// Treklarning ovozlarini baholash
var loudnessTrack1 = analyzeTrack(1, RMS_AMPLITUDE);
var loudnessTrack2 = analyzeTrack(2, RMS_AMPLITUDE);

// 1-trek ovozi 2-treknikidan balandligini tekshirish
// comparison1 degan mantiqiy ifodani yaratamiz
var comparison1 = (loudnessTrack1 > loudnessTrack2);
println("1-trek ovozi 2-treknikidan balandmi?");
println(comparison1);

// Har bir trekning ovozining balandligini 0.01ga taqqoslash uchun for halqasini yaratish
for (var track = 1; track < 3; track++) {
    var loudness = analyzeTrack(track, RMS_AMPLITUDE);
    println(track + "chi trek 0.01 dan balandmi?");
    println(loudness > 0.01);
}
