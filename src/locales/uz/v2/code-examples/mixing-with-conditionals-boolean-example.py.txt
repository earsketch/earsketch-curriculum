# Mantiqiy misol: Treklarimizning ovozini balandligin tahlil qilamiz

# Sozlash
from earsketch import *
setTempo(120)

# 2ta trekni yaratish
melody1 = RD_CINEMATIC_SCORE_STRINGS_14
melody2 = RD_UK_HOUSE__5THCHORD_1
fitMedia(melody1, 1, 1, 9)
fitMedia(melody2, 2, 1, 9)

# Treklarning ovozlarini baholash
loudnessTrack1 = analyzeTrack(1, RMS_AMPLITUDE)
loudnessTrack2 = analyzeTrack(2, RMS_AMPLITUDE)

# 1-trek ovozi 2-treknikidan balandligini tekshirish
# comparison1 degan mantiqiy ifodani yaratamiz
comparison1 = loudnessTrack1 > loudnessTrack2
print("1-trek ovozi 2-treknikidan balandmi?")
print(comparison1)

# Har bir trekning ovozining balandligini 0.01ga taqqoslash uchun for halqasini yaratish
for track in range(1, 3):
    loudness = analyzeTrack(track, RMS_AMPLITUDE)
    print(track + "chi trek 0.01 dan balandmi?")
    print(loudness > 0.01)
