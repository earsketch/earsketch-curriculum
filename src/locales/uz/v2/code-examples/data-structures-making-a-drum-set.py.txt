# Baraban to'plamini yaratish: Massivlarni makeBeat() bilan ishlatish

# Sozlash
from earsketch import *
setTempo(100)

# Bundan oldin biz har bir tovush uchun bitta trek ishlatardik (1-o'lcham):
kick = OS_KICK05
snare = OS_SNARE01
kickBeat = "0+++----0+++----"
snareBeat = "----0+++----0+++"
makeBeat(kick, 1, 1, kickBeat)
makeBeat(snare, 2, 1, snareBeat)

# Endi biz ularni birlashitarmiz (3-o'lcham):
drums = [OS_KICK05, OS_SNARE01]
beat = "0+++1+++0+++1+++"
makeBeat(drums, 1, 3, beat)
