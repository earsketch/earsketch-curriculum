# A-B-A Shakli: A va B qismli qo'shiq

# Sozlash
from earsketch import *
setTempo(120)

#Musiqa

# A qismini yaratish

fitMedia(RD_WORLD_PERCUSSION_KALIMBA_PIANO_7, 1, 1, 5)  # asosiy
fitMedia(RD_WORLD_PERCUSSION_DRUMPART_24, 2, 1, 5); # baraban
fitMedia(RD_UK_HOUSE__EVOLVEPAD_2, 3, 1, 5); # bass
setEffect(3, VOLUME, GAIN, -20, 1, 0, 5); # bass ovozini ko'tarish

# 5 va 9 o'lchamlar orasida 4 o'lchamli B qismini yaratish

fitMedia(RD_WORLD_PERCUSSION_KALIMBA_PIANO_1, 1, 5, 9); # asosiy melodiya o'zgarishi
fitMedia(RD_WORLD_PERCUSSION_DRUMPART_5, 2, 5, 9); # baraban o'zgarishi
fitMedia(RD_UK_HOUSE__EVOLVEPAD_3, 3, 5, 9); #kontrastli bass: sekinroq va ovosi balandroq
setEffect(3, VOLUME, GAIN, 5, 5, 5, 9); # bass ovozini ko'tarish
fitMedia(RD_WORLD_PERCUSSION_SEEDSRATTLE_1, 4, 5, 9); # shitirlash: yangi element qo'shiladi

# Endi 9 o'lchamda yana A qismiga qaytamiz

fitMedia(RD_WORLD_PERCUSSION_KALIMBA_PIANO_7, 1, 9, 13); # asosiy
fitMedia(RD_WORLD_PERCUSSION_DRUMPART_24, 2, 9, 13); # baraban
fitMedia(RD_UK_HOUSE__EVOLVEPAD_2, 3, 9, 13); # bass
setEffect(3, VOLUME, GAIN, 0, 9, -20, 13)  # bass ovozini ko'tarish
