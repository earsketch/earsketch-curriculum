# Murakkab zarflar: Effektlar zarfida o'zgartirishlar kiritish uchun trekda bir nechta setEffect() ni chaqirish

# Sozlash
from earsketch import *
setTempo(120)

# Musiqa
fitMedia(ELECTRO_ANALOGUE_LEAD_012, 1, 1, 9)

# Zarf vaqt nuqtalari (o'lchovlarda ko'rsatiladi)
pointA = 1
pointB = 4
pointC = 6.5
pointD = 7
pointE = 8.5
pointF = 9

setEffect(1, FILTER, FILTER_FREQ, 20, pointA, 10000, pointB); # Birinchi effekt, filtrdan qo'llanish

# Ikkinchi effekt, tovush balandligini o'zgartirish
setEffect(1, VOLUME, GAIN, -10, pointB, 0, pointC)  # Kreshendo, eng yuqori nuqta
setEffect(1, VOLUME, GAIN, 0, pointD, -10, pointE)  # pasayishning boshlanishi
setEffect(1, VOLUME, GAIN, -10, pointE, -60, pointF)  # Pasayishning tugashi
