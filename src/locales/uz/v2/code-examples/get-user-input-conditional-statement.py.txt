# Shartli bayonot: Funktsiyalarda o'zgarishlar yaratish uchun mantiqiy ifodalardan foydalanish

# Sozlash
from earsketch import *
setTempo(120)

melody1 = RD_WORLD_PERCUSSION_KALIMBA_PIANO_1
melody2 = RD_WORLD_PERCUSSION_KALIMBA_PIANO_2
drums1 = RD_WORLD_PERCUSSION_DRUMPART_24

# Funktsiya ta'rifi
def verse(start, variation):
    # o'zgarish qiymati rost yoki yolg'on bo'lishi mumkin
    if variation:
        fitMedia(melody1, 1, start, start + 4)
    else:
        fitMedia(melody2, 1, start, start + 4)
    fitMedia(drums1, 2, start, start + 4)  # bu shartli bayonotdan tashqarida (chekinish yo'q)

# Funktsiyani chaqirish
verse(1, True)
verse(7, False)
