// Shartli bayonot: Funktsiyalarda o'zgarishlar yaratish uchun mantiqiy ifodalardan foydalanish

// Sozlash
setTempo(120);

var melody1 = RD_WORLD_PERCUSSION_KALIMBA_PIANO_1;
var melody2 = RD_WORLD_PERCUSSION_KALIMBA_PIANO_2;
var drums1 = RD_WORLD_PERCUSSION_DRUMPART_24;

// Funktsiya ta'rifi
function verse(start, variation) {
    // o'zgarish qiymati rost yoki yolg'on bo'lishi mumkin
    if (variation) {
        fitMedia(melody1, 1, start, start + 4);
    } else {
        fitMedia(melody2, 1, start, start + 4);
    }
    fitMedia(drums1, 2, start, start + 4); // bu shartli bayonotdan tashqarida (checkinish yo'q)
}

// Funktsiyani chaqirish
verse(1, true);
verse(7, false);
